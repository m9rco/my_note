  #! /bin/bash

ssh_server(){
    arg="$*"
    echo '[*] Login ing ...'
    sleep 1s
    ssh $1@$2
}

su_ssh_server(){
    arg="$*"
    echo '[*] Login ing ...'
    sleep 1s
    sudo ssh $1@$2
}

su_ssh_by_private_key(){
    arg="$*"
    echo '[*] Login ing ...'
    sleep 1s
    sudo ssh -i $1 $2@$3
}
